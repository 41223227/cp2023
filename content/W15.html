void find_intersection(int x1, int y1, int x2, int y2, int cx, int cy, int cr) {
    float m, c, A, B, C, det, t;

    m = (float)(y2 - y1) / (x2 - x1);

    c = y1 - m * x1;

    A = 1 + m * m;
    B = -2 * cx + 2 * m * (c - cy);
    C = cx * cx + (c - cy) * (c - cy) - cr * cr;

    det = B * B - 4 * A * C;

    if (det >= 0) {
        t = (-B + sqrt(det)) / (2 * A);
        int x = x1 + t * (x2 - x1);
        int y = y1 + t * (y2 - y1);
        printf("Intersection point: (%d, %d)\n", x, y);
    }

    t = (-B - sqrt(det)) / (2 * A);
    int x = x1 + t * (x2 - x1);
    int y = y1 + t * (y2 - y1);
    printf("Intersection point: (%d, %d)\n", x, y);
}
